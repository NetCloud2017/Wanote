function quickSort(arr, left, right) {
	var len = arr.length,
		partitionIndex,
		left = typeof left != "number" ? 0 : left,
		right = typeof right != "number" ? len - 1 : right;

	if (left < right) {
		partitionIndex = partition(arr, left, right); // 5、 由于基值被调换，返回基值的位置， 作为区域分界

		///6、如此往复， 计算剩余分界排序。
		quickSort(arr, left, partitionIndex - 1);
		quickSort(arr, partitionIndex + 1, right);
	}
	return arr;
}

function partition(arr, left, right) {
	// 分区操作
	var pivot = left, // 1、 设定基准值（pivot）
		index = pivot + 1;
	for (var i = index; i <= right; i++) {
		if (arr[i] < arr[pivot]) {
			swap(arr, i, index); ///2、 假如 left  是 0， 第二个元素和 小于基值的元素调换位置
			index++; // 3、 原本的第二个元素被调换了所以用下一个元素。作为下次调换的对象。
		}
	}
	swap(arr, pivot, index - 1); //4、 基值与 最后一次被调换的值调换，得到本次最小的起始值。
	return index - 1;
}

function swap(arr, i, j) {
	// 位置替换
	var temp = arr[i];
	arr[i] = arr[j];
	arr[j] = temp;
}

function partition2(arr, low, high) {
	let pivot = arr[low];
	while (low < high) {
		while (low < high && arr[high] > pivot) {
			--high;
		}
		arr[low] = arr[high];
		while (low < high && arr[low] <= pivot) {
			++low;
		}
		arr[high] = arr[low];
	}
	arr[low] = pivot;
	return low;
}

function quickSort2(arr, low, high) {
	if (low < high) {
		let pivot = partition2(arr, low, high);
		quickSort2(arr, low, pivot - 1);
		quickSort2(arr, pivot + 1, high);
	}
	return arr;
}
